    ```dart
    /// App theme configuration for light and dark modes.
    class AppTheme {
      AppTheme._();

      /// Light theme data
      static ThemeData getLightTheme() {
        return ThemeData(
          colorScheme: ColorScheme.fromSeed(
            seedColor: Colors.blue,
            brightness: Brightness.light,
          ),
          useMaterial3: true,
          brightness: Brightness.light,

          // Accessibility: Enforce 48x48dp minimum touch targets
          elevatedButtonTheme: ElevatedButtonThemeData(

              minimumSize: const Size(48, 48),
            ),
          ),
          textButtonTheme: TextButtonThemeData(
            style: TextButton.styleFrom(
              minimumSize: const Size(48, 48),
            ),
          ),
          iconButtonTheme: IconButtonThemeData(
            style: IconButton.styleFrom(
              minimumSize: const Size(48, 48),
            ),
          ),
          floatingActionButtonTheme: FloatingActionButtonThemeData(
            extendedSizeConstraints: const BoxConstraints(
              minWidth: 48,
              minHeight: 48,
            ),
          ),
        );
      }

      /// Dark theme data
      static ThemeData getDarkTheme() {
        return ThemeData(
          colorScheme: ColorScheme.fromSeed(
            seedColor: Colors.blue,
            brightness: Brightness.dark,
          ),
          useMaterial3: true,
          brightness: Brightness.dark,

          // Accessibility: Enforce 48x48dp minimum touch targets
          elevatedButtonTheme: ElevatedButtonThemeData(
            style: ElevatedButton.styleFrom(
              minimumSize: const Size(48, 48),
            ),
          ),
          textButtonTheme: TextButtonThemeData(
            style: TextButton.styleFrom(

          floatingActionButtonTheme: FloatingActionButtonThemeData(

              minWidth: 48,

            ),
          ),
        );
      }
    }
    ```

    1. test/accessibility/touch_target_test.dart:
    ```dart
    import 'package:flutter/material.dart';
    import 'package:flutter_test/flutter_test.dart';
    import 'package:app_prenotazioni/main.dart';

    void main() {
      testWidgets('All buttons meet 48x48dp minimum touch target', (tester) async {
        await tester.pumpWidget(const MyApp());
        await tester.pumpAndSettle();

        // Find all buttons and icon buttons
        final buttons = find.byType(ElevatedButton);
        final iconButtons = find.byType(IconButton);
        final textButtons = find.byType(TextButton);
        final floatingActionButtons = find.byType(FloatingActionButton);

        // Test each button
        for (final button in [
          ...buttons.evaluate(),
          ...iconButtons.evaluate(),
          ...textButtons.evaluate(),
          ...floatingActionButtons.evaluate(),
        ]) {
          final widget = button.widget as Widget;

          // Get the size
          final size = tester.getSize(button);

          // Verify minimum size
          expect(
            size.width,
            greaterThanOrEqualTo(48),
            reason: '${widget.runtimeType} width must be at least 48dp',
          );
          expect(
            size.height,
            greaterThanOrEqualTo(48),
            reason: '${widget.runtimeType} height must be at least 48dp',
          );
        }
      });
    }
    ```

    2. test/accessibility/semantics_test.dart:
    ```dart
    import 'package:flutter/material.dart';
    import 'package:flutter_test/flutter_test.dart';
    import 'package:app_prenotazioni/main.dart';

    void main() {
      testWidgets('All interactive elements have semantic labels', (tester) async {
        await tester.pumpWidget(const MyApp());
        await tester.pumpAndSettle();

        // Find all interactive elements
        final inkWells = find.byType(InWell);
        final inkResponders = find.byType(InkResponse);

        // Check each interactive element has semantics
        for (final element in [...inkWells.evaluate(), ...inkResponders.evaluate()]) {
          final semantics = tester.getSemantics(element);

          // Verify the element has a label
          expect(
            semantics.label.isNotEmpty,
            true,
            reason: 'Interactive element must have a semantic label',
          );
        }
      });

      testWidgets('All form fields have semantic labels and hints', (tester) async {
        // Navigate to reservation form
        await tester.pumpWidget(const MyApp());
        await tester.pumpAndSettle();

        // Tap FAB to create new reservation
        final fab = find.byType(FloatingActionButton);
        await tester.tap(fab);
        await tester.pumpAndSettle();

        // Find all text fields
        final textFields = find.byType(TextField);

        for (final field in textFields.evaluate()) {
          final semantics = tester.getSemantics(field);

          // Verify the field has a label
          expect(
            semantics.label.isNotEmpty,
            true,
            reason: 'Text field must have a semantic label',
          );

          // Verify the field has a hint (helper text)
          expect(
            semantics.hint.isNotEmpty,
            true,
            reason: 'Text field should have a semantic hint',
          );
        }
      });
    }
    ```

    These automated tests verify accessibility compliance.
